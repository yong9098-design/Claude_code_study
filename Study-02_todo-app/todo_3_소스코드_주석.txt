    /**
     * 텍스트에서 카테고리 자동 감지
     * @param {string} text - 할일 입력 텍스트
     * @returns {string|null} 감지된 카테고리 코드 또는 null
     */
    function detectCategory(text) {
      const lower = text.toLowerCase();
      const scores = { work: 0, personal: 0, study: 0, club: 0 };  // 카테고리별 점수 초기화
      for (const [cat, keywords] of Object.entries(CATEGORY_KEYWORDS)) {
        for (const kw of keywords) {
          if (lower.includes(kw)) scores[cat]++;  // 키워드 포함 시 점수 +1
        }
      }
      const best = Object.entries(scores).sort((a, b) => b[1] - a[1])[0];
      return best[1] > 0 ? best[0] : null;  // 최고 점수가 0이면 감지 불가 → null
    }

     * JSON 파싱 오류 발생 시 빈 배열 반환 (안전 처리)

    function loadFromStorage() {
      try {
        return JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];
      } catch {
        return [];  // 파싱 오류 시 초기화
      }
    }

    /**
     * 할일 배열을 localStorage에 저장
     * @param {Array} arr - 저장할 tasks 배열
     */
    function saveToStorage(arr) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(arr));
    }

    /* ====================================================================

    /** 오늘 날짜를 "YYYY-MM-DD" 형식의 문자열로 반환 */
    function todayStr() {
      return new Date().toISOString().slice(0, 10);
    }

    /**
     * 완료일 문자열과 완료 여부를 받아 상태 코드 반환
     * @param {string|null} dateStr - "YYYY-MM-DD" 또는 null
     * @param {boolean} completed - 할일 완료 여부
     * @returns {'overdue'|'today'|'future'|'done'|null}
     */
    function dateStatus(dateStr, completed) {
      if (!dateStr) return null;      // 완료일 없으면 상태 없음
      if (completed) return 'done';   // 완료된 항목은 항상 'done'
      const today = todayStr();
      if (dateStr < today) return 'overdue';   // 오늘보다 이전 → 기한 초과
      if (dateStr === today) return 'today';   // 오늘 마감
      return 'future';                          // 아직 여유 있음
    }

    /**
     * 상태 코드를 한글 라벨로 변환
     * @param {string} status - dateStatus() 반환값
     * @returns {string} 화면에 표시할 한글 텍스트
     */
    function dateStatusLabel(status) {
      if (status === 'overdue') return '⚠ 기한 초과';
      if (status === 'today')   return '오늘까지';
      if (status === 'future')  return '예정';
      if (status === 'done')    return '완료됨';
      return '';
    }

       [JS-6] 할일 CRUD (Create / Read / Update / Delete) 함수

    /**
     * 새 할일 추가
     * @param {string} text - 할일 내용 (trim 처리)
     * @param {string} category - 카테고리 코드
     * @param {string|null} dueDate - 완료일 "YYYY-MM-DD" 또는 빈 문자열
     */
    function addTask(text, category, dueDate) {
      const trimmed = text.trim();
      if (!trimmed) { alert('할일 내용을 입력해주세요.'); return; }  // 공백 방지

      tasks.push({
        id: Date.now().toString(),   // 타임스탬프로 고유 ID 생성
        text: trimmed,
        category,
        completed: false,
        createdAt: Date.now(),
        dueDate: dueDate || null     // 빈 문자열은 null로 저장
      });
      saveToStorage(tasks);

      // 입력 필드 초기화 + 포커스 유지
      document.getElementById('task-input').value = '';
      document.getElementById('due-date-input').value = '';
      document.getElementById('task-input').focus();
    }

    /**
     * 할일 삭제
     * @param {string} id - 삭제할 Task의 id
     */
    function deleteTask(id) {
      tasks = tasks.filter(t => t.id !== id);
      saveToStorage(tasks);
    }

    /**
     * 완료 상태 토글 (완료 ↔ 미완료)
     * @param {string} id - 토글할 Task의 id
     */
    function toggleTask(id) {
      tasks = tasks.map(t => t.id === id ? { ...t, completed: !t.completed } : t);
      saveToStorage(tasks);
    }

    /**
     * 할일 텍스트 수정 (인라인 수정에서 호출)
     * @param {string} id - 수정할 Task의 id
     * @param {string} newText - 새 텍스트 (trim 처리)
     */
    function editTask(id, newText) {
      const trimmed = newText.trim();
      if (!trimmed) return;          // 빈 텍스트면 저장 안 함
      tasks = tasks.map(t => t.id === id ? { ...t, text: trimmed } : t);
      saveToStorage(tasks);
    }

    /**

     * @param {string|null} dateStr - "YYYY-MM-DD" 또는 null
     */
    function setDueDate(id, dateStr) {
      tasks = tasks.map(t => t.id === id ? { ...t, dueDate: dateStr || null } : t);
      saveToStorage(tasks);
    }

    /**

    function getFilteredTasks() {

    /* ====================================================================
       [JS-9] 진행률 대시보드 업데이트
       - 전체 완료율 + 기한 초과 카운트
       - 카테고리별 미니 진행 바 (work / personal / study / club)
       - 100% 달성 시 축하 메시지 + 초록 바
    ==================================================================== */
    function updateProgress() {
      const dash = document.getElementById('dashboard');

      // 할일이 하나도 없을 때
      if (tasks.length === 0) {
        dash.innerHTML = '<p class="empty-dashboard">할일을 추가해서 오늘을 계획해보세요! ✨</p>';
        return;
      }

      const total = tasks.length;
      const done  = tasks.filter(t => t.completed).length;
      const pct   = Math.round((done / total) * 100);       // 퍼센트 (반올림)
      const isComplete = pct === 100;

      // 기한 초과 건수 (미완료 & 완료일이 오늘보다 이전)
      const overdueCount = tasks.filter(t => !t.completed && t.dueDate && t.dueDate < todayStr()).length;
      const overdueHtml = overdueCount > 0
        ? `<span style="color:#E53935;font-size:0.82rem;font-weight:600;">⚠ 기한 초과 ${overdueCount}건</span>`
        : '';

      // 카테고리별 미니 진행 바 HTML 생성
      const catRows = ['work', 'personal', 'study', 'club'].map(cat => {
        const catAll  = tasks.filter(t => t.category === cat);
        const catDone = catAll.filter(t => t.completed).length;
        const catPct  = catAll.length ? Math.round((catDone / catAll.length) * 100) : 0;
        const label   = catAll.length ? `${catDone}/${catAll.length}` : '할일 없음';
        return `
          <div class="cat-progress">
            <div class="cat-progress-label">
              <span>${CAT_ICONS[cat]} ${CAT_NAMES[cat]}</span>
              <span>${label}</span>
            </div>
            <div class="mini-bar-wrap">
              <div class="mini-bar-fill ${cat}" style="width:${catPct}%"></div>
            </div>
          </div>`;
      }).join('');

      dash.innerHTML = `
        <div class="progress-main">
          <div class="progress-label">
            <span>전체 ${done}/${total} (${pct}%)</span>
            ${overdueHtml}
          </div>
          <div class="progress-bar-wrap">
            <div class="progress-bar-fill ${isComplete ? 'complete' : ''}" style="width:${pct}%"></div>
          </div>
          ${isComplete ? '<p class="congrats">🎉 모든 할일을 완료했습니다!</p>' : ''}
        </div>
        <div class="progress-categories">${catRows}</div>`;
    }

    /* ====================================================================
       [JS-10] 할일 목록 렌더링
       - 탭 카운터 업데이트
       - 빈 상태 메시지 처리
       - 각 할일을 <li> 카드로 생성 (draggable 포함)
       - 완료일 행 조건부 렌더링 (showDueDates 토글)
    ==================================================================== */
    function renderTasks() {
      const list = document.getElementById('task-list');
      const filtered = getFilteredTasks();

      // ── 탭 카운터 업데이트 (미완료 개수 배지) ──────────
      document.querySelectorAll('#category-tabs button').forEach(btn => {
        const filter = btn.dataset.filter;
        const pending = filter === 'all'
          ? tasks.filter(t => !t.completed).length
          : tasks.filter(t => t.category === filter && !t.completed).length;
        const base = filter === 'all' ? '전체' : `${CAT_ICONS[filter]} ${CAT_NAMES[filter]}`;
        btn.textContent = pending > 0 ? `${base} (${pending})` : base;
      });

      // ── 빈 상태 처리 ────────────────────────────────────
      if (filtered.length === 0) {
        const msg = currentFilter === 'all'
          ? '할일이 없습니다 🎉'
          : `${CAT_NAMES[currentFilter]} 카테고리에 할일이 없습니다.`;
        list.innerHTML = `<li class="empty-state">${msg}</li>`;
        updateProgress();
        updateClearBtn();
        return;
      }

      // ── 할일 카드 생성 ───────────────────────────────────
      list.innerHTML = '';
      filtered.forEach(task => {
        const li = document.createElement('li');
        li.className = `task-item${task.completed ? ' completed' : ''}`;
        li.dataset.id = task.id;

        // 완료일 상태 계산
        const status      = dateStatus(task.dueDate, task.completed);
        const statusLabel = dateStatusLabel(status);
        const dateClass   = status ? `task-date-input date-${status}` : 'task-date-input';

        li.draggable = true;  // 드래그 앤 드롭 활성화

        li.innerHTML = `
          <div class="task-main-row">
            <!-- 드래그 핸들: 이 아이콘을 잡고 드래그 -->
            <span class="drag-handle" title="드래그하여 순서 변경">⠿</span>
            <!-- 완료 체크박스 -->
            <input type="checkbox" ${task.completed ? 'checked' : ''}>
            <!-- 할일 텍스트 (기한 초과 시 .overdue 클래스로 빨간색) -->
            <span class="task-text${status === 'overdue' ? ' overdue' : ''}">${escapeHtml(task.text)}</span>
            <!-- 카테고리 배지 -->
            <span class="category-badge ${task.category}">${CAT_NAMES[task.category]}</span>
            <!-- 개별 삭제 버튼 -->
            <button class="btn-delete">🗑</button>
          </div>
          ${showDueDates ? `
          <div class="task-date-row">
            <label>📅 완료일</label>
            <input
              type="date"
              class="${dateClass}"
              value="${task.dueDate || ''}"
              data-role="due-date">
            ${task.dueDate ? `<button class="btn-clear-date" data-role="clear-date" title="완료일 삭제">✕</button>` : ''}
            ${statusLabel ? `<span class="date-status ${status}">${statusLabel}</span>` : ''}
          </div>` : ''}`;

        list.appendChild(li);
      });

      updateProgress();   // 진행률 대시보드 갱신
      updateClearBtn();   // 일괄 삭제 버튼 상태 갱신
    }

    /** 완료 항목 없으면 일괄 삭제 버튼 비활성화 */
    function updateClearBtn() {
      const btn = document.getElementById('clear-btn');
      btn.disabled = !tasks.some(t => t.completed);
    }

    /**
     * XSS 방지용 HTML 이스케이프
     * 사용자 입력 텍스트를 innerHTML에 삽입하기 전 반드시 호출
     */
    function escapeHtml(str) {
      return str
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;');
    }

    /* ====================================================================
       [JS-11] 인라인 수정 기능
       - .task-text 더블클릭 시 <span> → <input> 전환
       - Enter / blur: 저장 | Esc: 취소
       - saved 플래그로 Enter + blur 이벤트 중복 실행 방지
    ==================================================================== */
    function startEdit(li, id, currentText) {
      if (li.classList.contains('completed')) return;  // 완료 항목 수정 불가

      const span = li.querySelector('.task-text');
      const input = document.createElement('input');
      input.className = 'edit-input';
      input.value = currentText;
      input.maxLength = 100;
      span.replaceWith(input);  // span을 input으로 교체
      input.focus();
      input.select();           // 전체 텍스트 선택

      let saved = false;        // 중복 저장 방지 플래그

      function save() {
        if (saved) return;
        saved = true;
        editTask(id, input.value);
        renderTasks();
      }

      function cancel() {
        if (saved) return;
        saved = true;
        renderTasks();          // 저장 없이 원래 상태로 복원
      }

      input.addEventListener('keydown', e => {
        if (e.key === 'Enter') { e.preventDefault(); save(); }
        if (e.key === 'Escape') cancel();
      });
      input.addEventListener('blur', save);  // 포커스 벗어나면 자동 저장
    }

    /* ====================================================================
       [JS-12] 이벤트 리스너 등록
    ==================================================================== */

    // 추가 버튼 클릭
    document.getElementById('add-btn').addEventListener('click', () => {
      const text    = document.getElementById('task-input').value;
      const cat     = document.getElementById('category-select').value;

      renderTasks();
    });

    // 입력창 Enter 키 → 추가 버튼 클릭과 동일 동작
    document.getElementById('task-input').addEventListener('keydown', e => {
      if (e.key === 'Enter') document.getElementById('add-btn').click();
    });

    // 실시간 자동 카테고리 감지 (input 이벤트)
    document.getElementById('task-input').addEventListener('input', e => {
      const hint     = document.getElementById('auto-hint');
      const detected = detectCategory(e.target.value);
      if (detected) {
        document.getElementById('category-select').value = detected;  // 드롭다운 자동 변경
        hint.textContent = `✨ 자동 분류: ${CAT_ICONS[detected]} ${CAT_NAMES[detected]}`;
      } else {
        hint.textContent = '';
      }
    });

    // 이벤트 위임: #task-list 클릭 이벤트 (체크박스 / 삭제 / 완료일 삭제)
    document.getElementById('task-list').addEventListener('click', e => {
      const li = e.target.closest('[data-id]');  // 클릭된 요소의 가장 가까운 [data-id] 조상
      if (!li) return;

      const li = e.target.closest('[data-id]');

    /* ====================================================================
       [JS-13] 헤더 날짜/요일 실시간 표시
       - updateCurrentDate(): #current-date 영역에 "N월 N일 / X요일" 표시
       - setInterval로 1분마다 자동 갱신
    ==================================================================== */
    const DAY_NAMES = ['일요일', '월요일', '화요일', '수요일', '목요일', '금요일', '토요일'];

    function updateCurrentDate() {
      const now   = new Date();
      const month = now.getMonth() + 1;
      const date  = now.getDate();
      const day   = DAY_NAMES[now.getDay()];
      document.querySelector('#current-date .date-num').textContent = `${month}월 ${date}일`;
      document.querySelector('#current-date .date-day').textContent = day;
    }

    /* ====================================================================
       [JS-14] 드래그 앤 드롭으로 할일 순서 변경
       - dragstart: 드래그 시작한 아이템 ID 저장, .dragging 클래스 추가
       - dragend: 모든 드래그 관련 클래스 제거, dragSrcId 초기화
       - dragover: 마우스 위치(상반/하반부)에 따라 삽입 위치 표시선 토글
       - dragleave: 표시선 제거
       - drop: tasks 배열에서 순서 변경 후 저장 및 렌더링
    ==================================================================== */
    let dragSrcId = null;  // 현재 드래그 중인 아이템의 ID

    function initDragDrop() {
      const list = document.getElementById('task-list');

      // 드래그 시작
      list.addEventListener('dragstart', e => {
        const li = e.target.closest('.task-item');
        if (!li) return;
        dragSrcId = li.dataset.id;
        setTimeout(() => li.classList.add('dragging'), 0);  // 렌더링 후 적용 (비동기)
        e.dataTransfer.effectAllowed = 'move';
      });

      // 드래그 종료 — 모든 상태 클래스 정리
      list.addEventListener('dragend', () => {
        document.querySelectorAll('.task-item').forEach(item => {
          item.classList.remove('dragging', 'drag-over-top', 'drag-over-bottom');
        });
        dragSrcId = null;
      });

      // 드래그 오버 — 삽입 위치 표시선 갱신
      list.addEventListener('dragover', e => {
        e.preventDefault();  // 기본 드래그 금지 해제
        e.dataTransfer.dropEffect = 'move';
        const li = e.target.closest('.task-item');
        if (!li || li.dataset.id === dragSrcId) return;  // 자기 자신 위에서는 무시

        // 이전 표시선 모두 제거
        document.querySelectorAll('.task-item').forEach(item => {
          item.classList.remove('drag-over-top', 'drag-over-bottom');
        });

        // 마우스가 아이템의 상반부 → 위쪽에 선, 하반부 → 아래쪽에 선
        const rect = li.getBoundingClientRect();
        if (e.clientY < rect.top + rect.height / 2) {
          li.classList.add('drag-over-top');
        } else {
          li.classList.add('drag-over-bottom');
        }
      });

      // 드래그 떠남 — 표시선 제거
      list.addEventListener('dragleave', e => {
        const li = e.target.closest('.task-item');
        if (li) li.classList.remove('drag-over-top', 'drag-over-bottom');
      });

      // 드롭 — tasks 배열 순서 변경
      list.addEventListener('drop', e => {
        e.preventDefault();
        const targetLi = e.target.closest('.task-item');
        if (!targetLi || !dragSrcId || targetLi.dataset.id === dragSrcId) return;

        const targetId    = targetLi.dataset.id;
        const rect        = targetLi.getBoundingClientRect();
        const insertBefore = e.clientY < rect.top + rect.height / 2;  // 삽입 방향

        // 1. 소스 아이템을 배열에서 제거
        const srcIdx = tasks.findIndex(t => t.id === dragSrcId);
        const [removed] = tasks.splice(srcIdx, 1);

        // 2. 타겟 인덱스 재계산 (제거 후 인덱스 변동 반영)
        const tgtIdx = tasks.findIndex(t => t.id === targetId);

        // 3. 타겟 위/아래에 삽입
        tasks.splice(insertBefore ? tgtIdx : tgtIdx + 1, 0, removed);

        saveToStorage(tasks);
        renderTasks();
      });
    }

    /* ====================================================================
       [JS-15] 앱 초기화 — DOMContentLoaded
       HTML 파싱 완료 후 실행
    ==================================================================== */
    document.addEventListener('DOMContentLoaded', () => {
      updateCurrentDate();                      // 날짜/요일 즉시 표시
      setInterval(updateCurrentDate, 60000);    // 1분마다 날짜 갱신
      initDragDrop();                           // 드래그앤드롭 이벤트 등록
      renderTasks();                            // 저장된 할일 렌더링
      document.getElementById('task-input').focus();  // 입력창 자동 포커스
    });

  </script>
